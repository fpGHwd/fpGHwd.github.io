96. Unique Binary Search Trees

Solution 


class Solution {
  public:
    int numTrees(int n) {
        if (n == 0)
            return 1;
        if (n == 1)
            return 1;
        if (n == 2)
            return 2;
        int count = 0;
        for (int i = 1; i <= n; i++) {
            count = count + numTrees(i - 1) * numTrees(n - i);
        }
        return count;
    }
};
