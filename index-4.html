    901. Sort Array By Parity

    Solution 

    
class Solution {
  public:
    vector<int> sortArrayByParity(vector<int> &A) {
        sort(begin(A), end(A),
             [](auto &a, auto &b) { return a % 2 == 0 && b % 2 == 1; });
        return A;
    }
};
