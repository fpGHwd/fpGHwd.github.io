    最近在使用 org 的时候想把 org-archive-subtree 的条目和 org-capture-templates 那样，将每次当天 archived 的 headline 放到当日的日期 headline 下面，和日记一样组织。这样的话，将来查询的时候就可以顺便看一下当天发生的事情和当天完成的项目了。

    这个是日记的组织方式： 

    
(add-to-list 'org-capture-templates
                      '("zj" "journal" entry (file "/home/wd/dropbox/to-encfs/text/org/journal.org")
                        "* %u - %^{heading}\n  %?"))

    同时在 customize org-archive-location 中发现了有这种设定： 

    
"~/org/datetree.org::datetree/* Finished Tasks"
The "datetree/" string is special, signifying to archive
items to the datetree.  Items are placed in either the CLOSED
date of the item, or the current date if there is no CLOSED date.
The heading will be a subentry to the current date.  There doesn’t
need to be a heading, but there always needs to be a slash after
datetree.  For example, to store archived items directly in the
datetree, use "~/org/datetree.org::datetree/".

    可以把日记和归档的条目都放到一个文件，归档也不会经常去动，只需要看的时候搜索一下就行。

    92. Reverse Linked List II

    Solution 

    
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
  public:
    ListNode *reverseBetween(ListNode *head, int m, int n) {
        if (m == n) // 当 m = n 时并没有变化
            return head;

        int i = 1;
        ListNode *psuedo_head = new ListNode(), *pre = psuedo_head, *n_pre;

        for (ListNode *w = head; w;) {
            if (i <= m || i > n) {
                if (i == m)
                    n_pre = w;
                pre->next = w;
                if (i != m)
                    pre = w;
                w = w->next;
            } else if (i > m && i <= n) {
                ListNode *tmp = w;
                w = w->next;
                tmp->next = pre->next;
                pre->next = tmp;
                if (i == n)
                    pre = n_pre;
            }
            i++;
        }

        pre->next = nullptr; // 尾节点后续是 null

        return psuedo_head->next;
    }
};

class Solution {
  public:
    string toGoatLatin(string S) {
        int start = 0, end = 0, i = 1;
        string goat = "";
        while (start < S.length()) {
            if (i != 1)
                goat += " ";
            while (end < S.length() && isalpha(S[end]))
                end++;
            if (S[start] == 'a' || S[start] == 'e' || S[start] == 'i' ||
                S[start] == 'o' || S[start] == 'u' || S[start] == 'A' ||
                S[start] == 'E' || S[start] == 'I' || S[start] == 'O' ||
                S[start] == 'U') {
                goat += S.substr(start, end - start);
            } else {
                if (start + 1 != end) {
                    goat += S.substr(start + 1, end - start - 1);
                }
                goat += S.substr(start, 1);
            }
            goat += "ma";
            goat += string(i, 'a');

            end++, start = end, i++;
        }
        return goat;
    }
};

    572. Subtree of Another Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    bool isSubtree(TreeNode *s, TreeNode *t) {
        // return isSameTree(s, t) || isSameTree(s->left, t) ||
        //        isSameTree(s->right, t);
        if (s == NULL && t == NULL)
            return true;
        if (s == NULL)
            return false;
        if (t == NULL)
            return true;
        bool isSub;
        if (s->val == t->val)
            isSub = isSameTree(s, t);
        if (isSub)
            return true;
        return isSubtree(s->left, t) || isSubtree(s->right, t);
    }

    bool isSameTree(TreeNode *s, TreeNode *t) {
        if (s == nullptr && t == nullptr) {
            return true;
        } else {
            if (s == nullptr || t == nullptr)
                return false;
            if (s->val != t->val)
                return false;
            else
                return isSameTree(s->left, t->left) &&
                       isSameTree(s->right, t->right);
        }
    }
};

    93. Restore IP Addresses

    Solution 

    
class Solution {
  public:
    vector<string> addresses;
    vector<string> restoreIpAddresses(string s) {
        vector<string> path;
        dfs(s, path);
        return addresses;
    }

    void dfs(string s, vector<string> &path) {

        if (s == "" && path.size() == 4) {
            string tmp = "";
            for (string s : path) {
                if (tmp != "")
                    tmp += ".";
                tmp += s;
            }
            addresses.push_back(tmp);
            return;
        }

        if (path.size() > 4 || s == "")
            return;

        for (int i = 1; i <= s.length(); i++) {
            string ip = s.substr(0, i);
            if (i > 1 && s[0] == '0')
                return;
            if (stoi(ip) > 255)
                break;
            path.push_back(ip);
            dfs(s.substr(i, s.length() - i), path);
            path.pop_back();
        }
    }
};

    543. Diameter of Binary Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    int diameter = 0;
    int diameterOfBinaryTree(TreeNode *root) {
        depthTravel(root);
        return diameter;
    }
    int depthTravel(TreeNode *root) {
        if (root == nullptr)
            return 0;
        int left = depthTravel(root->left);
        int right = depthTravel(root->right);
        diameter = max(left + right, diameter);
        return 1 + max(left, right);
    }
};

Memory Usage: 21 MB, less than 13.09% of C++ submissions.

    501. Find Mode in Binary Search Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    vector<int> findMode(TreeNode *root) {
        if (root == NULL)
            return vector<int>{};
        vector<int> path, mode;
        travelMidOrder(root, path);
        int k, n = 0;
        for (int i = 0; i < path.size(); i = k) {
            k = i;
            while (k < path.size() && path[k] == path[i])
                k++;
            if (k - i > n) {
                mode.clear();
                mode.push_back(path[i]);
                n = k - i;
            } else if (k - i == n)
                mode.push_back(path[i]);
        }

        return mode;
    }

    void travelMidOrder(TreeNode *root, vector<int> &path) {
        if (root == nullptr)
            return;
        if (root->left)
            travelMidOrder(root->left, path);
        path.push_back(root->val);
        if (root->right)
            travelMidOrder(root->right, path);
    }
};

    733. Flood Fill

    Solution 

    
class Solution {
  public:
    vector<vector<int>>
    floodFill(vector<vector<int>> &image, int sr, int sc, int newColor) {
        int old_color = image[sr][sc];
        if (old_color == newColor)
            return image;

        floodFill(image, sr, sc, newColor, old_color);

        return image;
    }

    void floodFill(vector<vector<int>> &image,
                   int sr,
                   int sc,
                   int newColor,
                   int oldColor) {
        // paint
        image[sr][sc] = newColor;

        // paint others
        if (sr - 1 >= 0 && oldColor == image[sr - 1][sc])
            floodFill(image, sr - 1, sc, newColor);
        if (sr + 1 < image.size() && oldColor == image[sr + 1][sc])
            floodFill(image, sr + 1, sc, newColor);
        if (sc - 1 >= 0 && oldColor == image[sr][sc - 1])
            floodFill(image, sr, sc - 1, newColor);
        if (sc + 1 < image[0].size() && oldColor == image[sr][sc + 1])
            floodFill(image, sr, sc + 1, newColor);
    }
};
