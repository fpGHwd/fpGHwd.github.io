    linux 平台下 emacs native-comp 分支编译

    suzumiya natsuhi

    2020-11-26 09:53

    Comments

    • 编译 libgccjit

      1. 下载 gcc

        git clone git://gcc.gnu.org/git/gcc.git

        然后切换到目录 cd gcc

      2. 切换 gcc-9 分枝

        git checkout remotes/origin/releases/gcc-9

        创建 build 目录 mkdir -p build

        进入 build 目录 cd build

      3. configure

        ../configure --enable-host-shared --enable-languages=jit,c,c++,objc,obj-c++ --enable-checking=release --disable-bootstrap

      4. 编译

        make -j8

      5. 安装（默认到 /usr/local ）

        sudo make install

      6. 加载库

        sudo ldconfig

        测试是否能发现 libgccjit 库 sudo ldconfig -p | grep libgccjit

    • 编译 emacs

      1. 进入 emacs 目录后运行 autogen.sh

        ./autogen.sh

      2. 编译 emacs

        ./configure --with-nativecomp CC=clang

      3. 安装（默认 /usr/local ）

        sudo make install

    • 启动 doom emacs

      1. 启动的时候如果报错 comp-deferred-compilation-black-list

        然后找到这里解决：注释掉 .emacs.d/core/core.el 里面的一个过程 (add-to-list 'comp-deferred-compilation-black-list entry)

      2. 现在已经修复，直接 doom upgrade -f 并且 doom sync 最后 doom run --version 查看 emacs 版本即可。最后运行 doom run 。

    • 最后

      确实有流畅很多，也可能是心理作用吧。 :)

    Emacs Lisp Overlay

    suzumiya natsuhi

    2020-10-26 13:24

    Comments

    最近看到 Emacs Lisp 里面 Overlay 的概念，可以用于自定义介面 buffer 的文字的某些特殊效果。

    当然其实对文字的编辑和模式本身都可以是文字内容的一种展现方式，但是文字的属性（Text Properties）是文字的一部分，overlay 并不是文字的一部分。所以在 buffer 之间复制某些文字的时候，并不会同时复制 overlay ；同时改变 buffer 的文字属性其实也是在修改 buffer 本身。但 overlay 的改变并不会改变 buffer 本身，这种修改自然也不会被保存在这个 buffer 的 undo list 中。

    Org archive location

    最近在使用 org 的时候想把 org-archive-subtree 的条目和 org-capture-templates 那样，将每次当天 archived 的 headline 放到当日的日期 headline 下面，和日记一样组织。这样的话，将来查询的时候就可以顺便看一下当天发生的事情和当天完成的项目了。

    以下是日记的组织方式： 

    
(add-to-list 'org-capture-templates
                      '("zj" "journal" entry (file "/home/wd/dropbox/to-encfs/text/org/journal.org")
                        "* %u - %^{heading}\n  %?"))

    同时在 customize org-archive-location 中发现了有这种设定： 

    
"~/org/datetree.org::datetree/* Finished Tasks"
The "datetree/" string is special, signifying to archive
items to the datetree.  Items are placed in either the CLOSED
date of the item, or the current date if there is no CLOSED date.
The heading will be a subentry to the current date.  There doesn’t
need to be a heading, but there always needs to be a slash after
datetree.  For example, to store archived items directly in the
datetree, use "~/org/datetree.org::datetree/".

    可以把日记和归档的条目都放到一个文件，归档也不会经常去动，只需要看的时候搜索一下就行。

    92. Reverse Linked List II

    Solution 

    
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
  public:
    ListNode *reverseBetween(ListNode *head, int m, int n) {
        if (m == n) // 当 m = n 时并没有变化
            return head;

        int i = 1;
        ListNode *psuedo_head = new ListNode(), *pre = psuedo_head, *n_pre;

        for (ListNode *w = head; w;) {
            if (i <= m || i > n) {
                if (i == m)
                    n_pre = w;
                pre->next = w;
                if (i != m)
                    pre = w;
                w = w->next;
            } else if (i > m && i <= n) {
                ListNode *tmp = w;
                w = w->next;
                tmp->next = pre->next;
                pre->next = tmp;
                if (i == n)
                    pre = n_pre;
            }
            i++;
        }

        pre->next = nullptr; // 尾节点后续是 null

        return psuedo_head->next;
    }
};

    第一次做的时候没想那么多，暴力的做法。后来做的时候就想到这是将链表的导致的前插和尾插组合在一起的题。并不难，只要分别搞清楚这两种写法就能写出来。

    96. Unique Binary Search Trees

    Solution 

    class Solution {
  public:
    int numTrees(int n) {
        if (n == 0 || n == 1)return 1;

        int count = 0;
        for (int i = 1; i <= n; i++)count += numTrees(i - 1) * numTrees(n - i);

        return count;
    }
};

    824. Goat Latin

    Solution 

    
class Solution {
  public:
    string toGoatLatin(string S) {
        int start = 0, end = 0, i = 1;
        string goat = "";
        while (start < S.length()) {
            if (i != 1)
                goat += " ";
            while (end < S.length() && isalpha(S[end]))
                end++;
            if (S[start] == 'a' || S[start] == 'e' || S[start] == 'i' ||
                S[start] == 'o' || S[start] == 'u' || S[start] == 'A' ||
                S[start] == 'E' || S[start] == 'I' || S[start] == 'O' ||
                S[start] == 'U') {
                goat += S.substr(start, end - start);
            } else {
                if (start + 1 != end) {
                    goat += S.substr(start + 1, end - start - 1);
                }
                goat += S.substr(start, 1);
            }
            goat += "ma";
            goat += string(i, 'a');

            end++, start = end, i++;
        }
        return goat;
    }
};
