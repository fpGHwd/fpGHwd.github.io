    1. Diameter of Binary Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    int diameter = 0;
    int diameterOfBinaryTree(TreeNode *root) {
        depthTravel(root);
        return diameter;
    }
    int depthTravel(TreeNode *root) {
        if (root == nullptr)
            return 0;
        int left = depthTravel(root->left);
        int right = depthTravel(root->right);
        diameter = max(left + right, diameter);
        return 1 + max(left, right);
    }
};

    501. Find Mode in Binary Search Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    vector<int> findMode(TreeNode *root) {
        if (root == NULL)
            return vector<int>{};
        vector<int> path, mode;
        travelMidOrder(root, path);
        int k, n = 0;
        for (int i = 0; i < path.size(); i = k) {
            k = i;
            while (k < path.size() && path[k] == path[i])
                k++;
            if (k - i > n) {
                mode.clear();
                mode.push_back(path[i]);
                n = k - i;
            } else if (k - i == n)
                mode.push_back(path[i]);
        }

        return mode;
    }

    void travelMidOrder(TreeNode *root, vector<int> &path) {
        if (root == nullptr)
            return;
        if (root->left)
            travelMidOrder(root->left, path);
        path.push_back(root->val);
        if (root->right)
            travelMidOrder(root->right, path);
    }
};

    733. Flood Fill

    Solution 

    
class Solution {
  public:
    vector<vector<int>>
    floodFill(vector<vector<int>> &image, int sr, int sc, int newColor) {
        int old_color = image[sr][sc];
        if (old_color == newColor)
            return image;

        floodFill(image, sr, sc, newColor, old_color);

        return image;
    }

    void floodFill(vector<vector<int>> &image,
                   int sr,
                   int sc,
                   int newColor,
                   int oldColor) {
        // paint
        image[sr][sc] = newColor;

        // paint others
        if (sr - 1 >= 0 && oldColor == image[sr - 1][sc])
            floodFill(image, sr - 1, sc, newColor);
        if (sr + 1 < image.size() && oldColor == image[sr + 1][sc])
            floodFill(image, sr + 1, sc, newColor);
        if (sc - 1 >= 0 && oldColor == image[sr][sc - 1])
            floodFill(image, sr, sc - 1, newColor);
        if (sc + 1 < image[0].size() && oldColor == image[sr][sc + 1])
            floodFill(image, sr, sc + 1, newColor);
    }
};

class Solution {
  public:
    int countBinarySubstrings(string s) {
        if (s == "")
            return 0;
        vector<int> vi_pos;
        for (int i = 1; i < s.length(); i++) {
            if (s[i] != s[i - 1])
                vi_pos.push_back(i);
        }
        int count = 0, left, right;
        for (int i = 0; i < vi_pos.size(); i++) {
            left = vi_pos[i] - (i == 0 ? 0 : vi_pos[i - 1]);
            right = ((i == vi_pos.size() - 1) ? s.length() : vi_pos[i + 1]) -
                    vi_pos[i];
            count += min(left, right);
        }
        return count;
    }
};

    687. Longest Univalue Path

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */

class Solution {
  public:
    int luvp = 0;
    int longestUnivaluePath(TreeNode *root) {
        if (root == nullptr)
            return 0;
        travelPath(root);
        return luvp - 1;
    }

    int travelPath(TreeNode *root) {
        int left = 0, right = 0;
        if (root->left != nullptr)
            left = travelPath(root->left);
        if (root->right != nullptr)
            right = travelPath(root->right);
        int left_tmp =
            ((root->left && root->left->val == root->val) ? left : 0);
        int right_tmp =
            ((root->right && root->right->val == root->val) ? right : 0);
        luvp = max(luvp, 1 + left_tmp + right_tmp);
        return 1 + max(left_tmp, right_tmp);
    }
};


    
Status: Accepted (71/71)

Runtime: 184 ms, faster than 86.14% of C++ submissions.

Memory Usage: 72.6 MB, less than 5.21% of C++ submissions.

    第一次做的时候没有做出来，觉得有点复杂。不过说到底还只是一个简单的树遍历，当然可以认为和求树深度的那类题其实差不多。其中每个点的处理就是找到是否比现有的最长路径 luvp 更长的 1 + left_tmp + right_tmp ，然后返回的若是 1 + left_tmp + right_tmp ，那么得出来的结果不是 最长的path ，而是相同值最多的相邻点个数。所以最后修改一下，返回稍长的路径即可。

    树的题，递归是最简洁的。

    599. Minimum Index Sum of Two Lists

    Solution 

    
class Solution {
  public:
    vector<string> findRestaurant(vector<string> &list1,
                                  vector<string> &list2) {
        unordered_map<string, int> umsi;
        for (int i = 0; i < list1.size(); i++) {
            umsi[list1[i]] = i;
        }

        int idx_sum = INT_MAX;
        vector<string> vsr;
        for (int i = 0; i < list2.size(); i++) {
            if (umsi.find(list2[i]) != umsi.end()) {
                if (idx_sum > i + umsi[list2[i]]) {
                    vsr.clear();
                    vsr.push_back(list2[i]);
                    idx_sum = i + umsi[list2[i]];
                } else if (idx_sum == i + umsi[list2[i]]) {
                    vsr.push_back(list2[i]);
                }
            }
        }

        return vsr;
    }
};

    617. Merge Two Binary Trees

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    TreeNode *mergeTrees(TreeNode *t1, TreeNode *t2) {
        if (t1 == NULL && t2 == NULL)
            return NULL;
        TreeNode *t3 = new TreeNode((t1 == NULL ? 0 : t1->val) +
                                    (t2 == NULL ? 0 : t2->val));
        t3->left = mergeTrees(t1 == NULL ? NULL : t1->left,
                              t2 == NULL ? NULL : t2->left);
        t3->right = mergeTrees(t1 == NULL ? NULL : t1->right,
                               t2 == NULL ? NULL : t2->right);
        return t3;
    }
};

/*
// Definition for Employee.
class Employee {
public:
    int id;
    int importance;
    vector<int> subordinates;
};
*/

class Solution {
  public:
    int getImportance(vector<Employee *> employees, int id) {

        unordered_map<int, Employee *> umii;
        for (Employee *e : employees) {
            umii[e->id] = e;
        }

        int importance = 0;
        queue<int> qi;
        qi.push(id);

        int tmp;
        while (!qi.empty()) {
            tmp = qi.front();
            importance += umii[tmp]->importance;
            qi.pop();
            for (int i : umii[tmp]->subordinates)
                qi.push(i);
        }

        return importance;
    }
};

    671. Second Minimum Node In a Binary Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    int fm = -1, sm = -1;
    int findSecondMinimumValue(TreeNode *root) {
        if (root == NULL)
            return sm;

        if (fm == -1)
            fm = root->val;
        else {
            if (sm == -1) {
                if (root->val > fm)
                    sm = root->val;
                else if (root->val < fm) {
                    sm = fm;
                    fm = root->val;
                } else {
                    // do nothing
                }
            } else {
                if (root->val > sm) {
                    // do noting
                } else if (root->val < fm) {
                    sm = fm;
                    fm = root->val;
                } else if (root->val == sm || root->val == fm) {
                    // do nothing
                } else { // root->val > fm and root->val < sm;
                    sm = root->val;
                }
            }
        }

        findSecondMinimumValue(root->left);
        findSecondMinimumValue(root->right);

        return sm;
    }
};
