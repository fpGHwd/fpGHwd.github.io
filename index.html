    687. Longest Univalue Path

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */

class Solution {
  public:
    int luvp = 0;
    int longestUnivaluePath(TreeNode *root) {
        if (root == nullptr)
            return 0;
        travelPath(root);
        return luvp - 1;
    }

    int travelPath(TreeNode *root) {
        int left = 0, right = 0;
        if (root->left != nullptr)
            left = travelPath(root->left);
        if (root->right != nullptr)
            right = travelPath(root->right);
        int left_tmp =
            ((root->left && root->left->val == root->val) ? left : 0);
        int right_tmp =
            ((root->right && root->right->val == root->val) ? right : 0);
        luvp = max(luvp, 1 + left_tmp + right_tmp);
        return 1 + max(left_tmp, right_tmp);
    }
};

    第一次做的时候没有做出来，觉得有点复杂。不过说到底还只是一个简单的树遍历，当然可以认为和求树深度的那类题其实差不多。其中每个点的处理就是找到是否比现有的最长路径 luvp - 1 更长的 1 + left_tmp + right_tmp ，然后返回的若是 1 + left_tmp + right_tmp ，那么得出来的结果不是 最长的path ，而是相同值最多的相邻点个数。所以最后修改一下，返回稍长的路径即可。

    树的题，递归是最简洁的。

    599. Minimum Index Sum of Two Lists

    Solution 

    
class Solution {
  public:
    vector<string> findRestaurant(vector<string> &list1,
                                  vector<string> &list2) {
        unordered_map<string, int> umsi;
        for (int i = 0; i < list1.size(); i++) {
            umsi[list1[i]] = i;
        }

        int idx_sum = INT_MAX;
        vector<string> vsr;
        for (int i = 0; i < list2.size(); i++) {
            if (umsi.find(list2[i]) != umsi.end()) {
                if (idx_sum > i + umsi[list2[i]]) {
                    vsr.clear();
                    vsr.push_back(list2[i]);
                    idx_sum = i + umsi[list2[i]];
                } else if (idx_sum == i + umsi[list2[i]]) {
                    vsr.push_back(list2[i]);
                }
            }
        }

        return vsr;
    }
};

    617. Merge Two Binary Trees

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    TreeNode *mergeTrees(TreeNode *t1, TreeNode *t2) {
        if (t1 == NULL && t2 == NULL)
            return NULL;
        TreeNode *t3 = new TreeNode((t1 == NULL ? 0 : t1->val) +
                                    (t2 == NULL ? 0 : t2->val));
        t3->left = mergeTrees(t1 == NULL ? NULL : t1->left,
                              t2 == NULL ? NULL : t2->left);
        t3->right = mergeTrees(t1 == NULL ? NULL : t1->right,
                               t2 == NULL ? NULL : t2->right);
        return t3;
    }
};

    690. Employee Importance 

    
/*
// Definition for Employee.
class Employee {
public:
    int id;
    int importance;
    vector<int> subordinates;
};
*/

class Solution {
  public:
    int getImportance(vector<Employee *> employees, int id) {

        unordered_map<int, Employee *> umii;
        for (Employee *e : employees) {
            umii[e->id] = e;
        }

        int importance = 0;
        queue<int> qi;
        qi.push(id);

        int tmp;
        while (!qi.empty()) {
            tmp = qi.front();
            importance += umii[tmp]->importance;
            qi.pop();
            for (int i : umii[tmp]->subordinates)
                qi.push(i);
        }

        return importance;
    }
};

    671. Second Minimum Node In a Binary Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    int fm = -1, sm = -1;
    int findSecondMinimumValue(TreeNode *root) {
        if (root == NULL)
            return sm;

        if (fm == -1)
            fm = root->val;
        else {
            if (sm == -1) {
                if (root->val > fm)
                    sm = root->val;
                else if (root->val < fm) {
                    sm = fm;
                    fm = root->val;
                } else {
                    // do nothing
                }
            } else {
                if (root->val > sm) {
                    // do noting
                } else if (root->val < fm) {
                    sm = fm;
                    fm = root->val;
                } else if (root->val == sm || root->val == fm) {
                    // do nothing
                } else { // root->val > fm and root->val < sm;
                    sm = root->val;
                }
            }
        }

        findSecondMinimumValue(root->left);
        findSecondMinimumValue(root->right);

        return sm;
    }
};

    819. Most Common Word

    Solution 

    
class Solution {
  public:
    string mostCommonWord(string paragraph, vector<string> &banned) {
        unordered_map<string, int> umsi;
        int start, end;
        transform(paragraph.begin(),paragraph.end(),paragraph.begin(),::tolower);
        for (int i = 0; i < paragraph.length(); i++) {
            while (i < paragraph.length() && !isalpha(paragraph[i]))
                i++;
            start = i;
            while (i < paragraph.length() && isalpha(paragraph[i]))
                i++;
            end = i;
            umsi[paragraph.substr(start, end - start)]++;
        }

        for (string s : banned)
            if (umsi.find(s) != umsi.end())
                umsi.erase(s);

        string w = "";
        for (auto it : umsi) {
            if (w == "")
                w = it.first;
            else if (umsi[it.first] > umsi[w])
                w = it.first;
        }

        return w;
    }
};

    107. Binary Tree Level Order Traversal II

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    vector<vector<int>> levelOrderBottom(TreeNode *root) {
        vector<vector<int>> result;

        queue<TreeNode *> q;
        TreeNode *last, *end;
        if (root) {
            q.push(root);
            end = root;
            last = end;
            result.push_back(vector<int>{});
        }
        while (!q.empty()) {
            TreeNode *temp = q.front();
            result[result.size() - 1].push_back(temp->val);
            q.pop();
            if (temp->left)q.push(temp->left);
            if (temp->right)q.push(temp->right);
            if (temp == last && !q.empty())result.push_back(vector<int>{}); // 达到一层结尾还有元素就增加一个空 vector
            if (temp == last)last=q.back();
        }
        reverse(result.begin(), result.end());
        return result;
    }
};

    108. Convert Sorted Array to Binary Search Tree

    Solution 

    
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
  public:
    TreeNode *sortedArrayToBST(vector<int> &nums) {
        // create tree
        return toHeightBST(nums, 0, nums.size() - 1);
    }

    TreeNode *toHeightBST(vector<int> &nums, int l, int r) {
        if (l > r)
            return NULL;
        int mid = l + (r - l) / 2;
        TreeNode *root = new TreeNode(nums[mid]);
        root->left = toHeightBST(nums, l, mid - 1);
        root->right = toHeightBST(nums, mid + 1, r);
        return root;
    }
};

    一开始以为是树的高度的调整题，后来发现这个和 用数组表示的树构造一棵树 的做法差不多。

    937. Reorder Data in Log Files

    Solution 

    
class Solution {
    public:
        vector<string> reorderLogFiles(vector<string> &logs) {
            stable_sort(begin(logs), end(logs), [](auto &a, auto &b) {
                auto A = 1 + a.find(' '), B = 1 + b.find(' ');
                return a[A] > '@' &&
                    (b[B] < '@' ||
                     a.substr(A).append(a, 0, A) < b.substr(B).append(b, 0, B));
            });
            return logs;
        }
};

    先找到 identifier 和后面部分的分隔点 A 和 B ，然后用 '@' 作为判断后面是数字还是字母的基准。ASCII 码表中， '0' 和 '9' 分别是 48 和 57 ， '@' 是 64 ， 'a' 和 'z' 分别是 97 和 122 。 return 后的部分同时保证了 letter-logs 排在 digit-logs 的前面；后部分（数字或者字母）拼接 identifier 后用字典序排列，也即后部分如果恰好相同时， identifier dominating。

    这个写法是翻解答时看到的，比较简洁，如果熟悉 sort 的话也比较容易懂。sort 的关键还是后面的函数 template< class RandomIt, class Compare > void stable_sort( RandomIt first, RandomIt last, Compare comp ); 。
